# Criar uma coluna de Receita (Quantidade * Valor Unitrio)
df['Receita'] = df['Quantidade'] * df['Valor Unitrio']

# Agrupar os dados por Categoria e calcular Receita total e Quantidade total
categoria_summary = df.groupby('Categoria').agg(
    Receita_Total=('Receita', 'sum'),
    Quantidade_Total=('Quantidade', 'sum')
).reset_index()

# Calcular o ticket mdio (Receita Total / Quantidade Total)
categoria_summary['Ticket_Medio'] = categoria_summary['Receita_Total'] / categoria_summary['Quantidade_Total']

# Ordenar as categorias pelo ticket mdio em ordem decrescente
categoria_summary = categoria_summary.sort_values(by='Ticket_Medio', ascending=False)

# Exibir o resultado
print("Categorias com Ticket Mdio:")
